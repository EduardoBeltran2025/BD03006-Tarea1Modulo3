SELECT
    p.ProductID,
    s.CustomerStateID,
    p.ProductName,
    p.ProductSKU,
    p.KitType,
    p.RetailPrice,
    s.OrderDate,
    s.ShipDate,
	s.Quantity
FROM
    dbo.Sales AS s
JOIN
    dbo.Product AS p ON s.ProductID = p.ProductID;

SELECT
    S.StateID,
    S.StateCode,
    S.RegionID,
    S.StateName,
    PS.OrderDate,
    PS.ShipDate,
    PS.CustomerStateID
FROM
    State AS S
LEFT JOIN
    Sales AS PS ON S.StateID = PS.CustomerStateID;
